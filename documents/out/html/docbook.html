Описание продукта

Программный продукт Игра пинг-понг был разработан для развлечения. Игра содержит в себе поле с шариком и двумя ракетками. Игра предназначена как для одного, так и для двоих игроков, в игре так же присутсвует демо-режим где можно посмотреть игру компьюетра.

Первый запуск

При первом запуске игры пользователю предоставлено поле и предлагается выбрать режим игры, код выбора представлен ниже:

    /**
    * Start
    *
    * Sets ball and pads
    *
    * @param Int $numPlayers
    * @return null
    */
    start: function(numPlayers) {
        if (!this.playing) {
            this.scores = [0, 0];
            this.playing = true;
            this.leftPaddle.setAuto(numPlayers < 1, this.level(0));
            this.rightPaddle.setAuto(numPlayers < 2, this.level(1));
            this.ball.reset();
            this.runner.hideCursor();
        }
    },

Обновление состояния игры приведено в программном листинге ниже:

    /**
    * Update
    *
    * Updates the game according to the DeltaTime
    *
    * @param Int $dt
    * @return null
    */
    update: function(dt) {
        this.leftPaddle.update(dt, this.ball);
        this.rightPaddle.update(dt, this.ball);
        if (this.playing) {
            var dx = this.ball.dx;
            var dy = this.ball.dy;
            this.ball.update(dt, this.leftPaddle, this.rightPaddle);
            if (this.ball.dx < 0 && dx > 0)
                this.sounds.ping();
            else if (this.ball.dx > 0 && dx < 0)
                this.sounds.pong();
            else if (this.ball.dy * dy < 0)
                this.sounds.wall();

            if (this.ball.left > this.width)
                this.goal(0);
            else if (this.ball.right < 0)
                this.goal(1);
        }
    },
